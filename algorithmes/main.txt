main
VARIABLE
    jeu : Jeu
    ok : bool
    cmd : char
DEBUT
    // Charger la configuration générale
    ok ← charger_configuration("config.txt", jeu)

    si !ok alors
        écrire"Erreur : config.txt introuvable ou invalide"
    fin si

    // Charger carte 
    ok ← charger_carte("level.txt", jeu)
    si !ok alors
        écrire "erreur : level.txt introuvable ou invalide"
    fin si

    // Initialisation du jeu (joueur + monstres + brouillard)
    initialiser_jeu(jeu)

    // Boucle principale
    tant que !jeu.etat_termine faire
        afficher_jeu(jeu)
        saisir_commande(cmd)

        traiter_commande(cmd, jeu)
        mettre_a_jour_monstres(jeu)
        mettre_a_jour_visibilite(jeu)
        verifier_conditions_victoire_defaite(jeu)
    fin tant que

    // Fin de jeu
    afficher_game_over(jeu)
FIN
