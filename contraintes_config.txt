CONST TAILLE_MAX = 1840

// --- Structures de listes génériques ---
TYPE Liste_id_contraintes = ENREGISTREMENT
    nb : ENTIER                                // Nombre d’identifiants dans la liste
    ids : TABLEAU[TAILLE_MAX] D'ENTIER        // Tableau des identifiants (entiers)
FIN ENREGISTREMENT

TYPE Liste_Id_Equipements = ENREGISTREMENT
    taille : ENTIER                           // Nombre d’éléments dans la liste
    ids : TABLEAU[TAILLE_MAX] D'ENTIER        // Tableau contenant les identifiants d’équipements
FIN ENREGISTREMENT

// --- Structures de configuration ---
TYPE Config_item = ENREGISTREMENT
    id : ENTIER
    symbole : CARACTÈRE
    nom : CHAINE
    description : CHAINE
    bonus : TABLEAU[6] D'ENTIER
FIN ENREGISTREMENT

TYPE Config_monstre = ENREGISTREMENT
    id : ENTIER
    symbole : CARACTÈRE
    nom : CHAINE
    description : CHAINE
    stats_base : TABLEAU[6] D'ENTIER
    typeIA : ENTIER
    id_item_contrainte : ENTIER
    contrainte_stats : TABLEAU[6] D'ENTIER
    inventaire_ids : Liste_Id_Equipements      // Liste d’ids d’équipements possédés par le monstre
FIN ENREGISTREMENT

TYPE Config_porte = ENREGISTREMENT
    id : ENTIER
    symbole : CARACTÈRE
    id_item_contrainte : ENTIER                // Id de l’équipement requis pour la porte
    contrainte_stats : TABLEAU[6] D'ENTIER
FIN ENREGISTREMENT

TYPE Config_joueur = ENREGISTREMENT
    id : ENTIER
    symbole : CARACTÈRE
    nom : CHAINE
    description : CHAINE
    stats : TABLEAU[6] D'ENTIER
    inventaire_ids : Liste_Id_Equipements      // Liste d’ids d’équipements possédés par le joueur
    actif : BOOLEEN
FIN ENREGISTREMENT

TYPE Config_conditions_jeu = ENREGISTREMENT
    contraintes : TABLEAU[TAILLE_MAX] DE Contrainte  // Toutes les contraintes possibles (à définir selon besoin réel)
    nbContraintes : ENTIER
    victoire : Liste_id_contraintes           // Liste des identifiants correspondant aux conditions de victoire
    defaite : Liste_id_contraintes            // Liste des identifiants correspondant aux conditions de défaite
FIN ENREGISTREMENT

// --- Structures d’entités dynamiques ---
TYPE Joueur = ENREGISTREMENT
    x : ENTIER
    y : ENTIER
    inventaire : TABLEAU[TAILLE_MAX] D'ENTIER
    nb_inventaire : ENTIER
    stat : TABLEAU[6] D'ENTIER
FIN ENREGISTREMENT

TYPE Monstre = ENREGISTREMENT
    x : ENTIER
    y : ENTIER
    idConfig : ENTIER
    stats : TABLEAU[6] D'ENTIER
    actif : BOOLEEN
FIN ENREGISTREMENT

TYPE Carte = ENREGISTREMENT
    largeur : ENTIER
    hauteur : ENTIER
    cases : TABLEAU[23][80] DE CARACTERE
    visible : TABLEAU[23][80] DE BOOLEEN     // Pour le fog of war
    actif : BOOLEEN
FIN ENREGISTREMENT

// --- Structure principale du jeu ---
TYPE Jeu = ENREGISTREMENT
    carte : Carte
    joueur : Joueur
    cfg_joueur : Config_joueur
    cfg_items : TABLEAU[26] DE Config_item
    nb_cfg_items : ENTIER
    cfg_monstres : TABLEAU[26] DE Config_monstre
    nb_cfg_monstres : ENTIER
    cfg_portes : TABLEAU[10] DE Config_porte
    nb_cfg_portes : ENTIER
    cfgConditions : Config_conditions_jeu
    cfgSymboles : TABLE_CONFIG_SYMBOLES      // Définir selon usage réel
    monstres : TABLEAU[TAILLE_MAX] DE Monstre
    nb_monstres : ENTIER
    etat_termine : BOOLEEN
    victoire : BOOLEEN
FIN ENREGISTREMENT

TYPE Contrainte = ENREGISTREMENT

id : ENTIER
description : CHAINE
stats_min : TABLEAU[6] D'ENTIER

FIN ENREGISTREMENT

Rôle : Représente une contrainte de jeu. 
- id : identifiant unique de la contrainte
- description : texte explicatif (ex. "Force ≥ 5 pour ouvrir la porte")
- stats_min : tableau de 6 entiers indiquant les valeurs minimales requises 
              pour les statistiques du joueur (comparées aux stats du joueur).



// --- Exemple de fonction de vérification des contraintes ---
FONCTION verifier_contraintes(possede : Liste_id_contraintes, requis : Liste_id_contraintes) → BOOLEEN
    SI possede.nb < requis.nb ALORS
        RETOURNER FAUX
    FIN SI
    POUR i DE 0 À requis.nb - 1 FAIRE
        SI possede.ids[i] < requis.ids[i] ALORS
            RETOURNER FAUX
        FIN SI
    FIN POUR
    RETOURNER VRAI
FIN FONCTION
